üöÄ Quick Start

npm install @awspirin/policy-generator
import { PolicyGenerator } from '@awspirin/policy-generator';

function App() {
  const myARNs = [
    {
      arn: 'arn:aws:s3:::my-bucket/*',
      service: 's3',
      displayName: 'My Bucket'
    },
    {
      arn: 'arn:aws:ec2:*:*:instance/*',
      service: 'ec2',
      displayName: 'EC2 Instances'
    }
  ];

  return (
    <PolicyGenerator
      arnList={myARNs}
      arnInputMode="auto"
      layout="responsive"
      onChange={(policy) => console.log(policy)}
    />
  );
}
